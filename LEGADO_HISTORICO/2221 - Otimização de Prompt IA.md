  "D:\backups",

  "C:\Users\felip\Documents",

$jobs = foreach ($file in $uniqueFiles) {
  Start-Job -ScriptBlock {
    param($filePath)
    try {
      $hash = Get-FileHash -Path $filePath -Algorithm SHA256
      $info = Get-Item -Path $filePath
      return @{
        Path = $filePath
        SHA256 = $hash.Hash
        SizeKB = [math]::Round($info.Length / 1KB, 2)
        Modified = $info.LastWriteTime
      }
    } catch {
      return $null
    }
  } -ArgumentList $_.FullName
}

# Coletar resultados
$jobData = @()
foreach ($job in $jobs) {
  $out = Receive-Job -Job $job -Wait -ErrorAction SilentlyContinue
  if ($out) {
    $results[$out.Path] = $out
    $jobData += $out
  }
  Remove-Job $job
}

# Salvar novo cache
$results | ConvertTo-Json -Depth 5 | Set-Content -Path $cacheFile -Encoding UTF8

function Get-CoreId {

function Read-Json([string]$path){
  $raw = Get-Content -LiteralPath $path -Raw -Encoding UTF8
  # PowerShell 7+: -AsHashtable ensures deterministic traversal ordering by our construction (ordered hashtables).
  try {

function Write-Bytes([byte[]]$bytes, [string]$dest){
  $dir = Split-Path -Parent $dest
  if(-not (Test-Path -LiteralPath $dir)){ New-Item -ItemType Directory -Path $dir -Force | Out-Null }
  [System.IO.File]::WriteAllBytes($dest, $bytes)
}

function Hex-To-Bytes([string]$hex){
  $clean = ($hex -replace '\s','')
  if(($clean.Length % 2) -ne 0){ throw "write.hex: hex length must be even" }
  $arr = New-Object byte[] ($clean.Length/2)
  for($i=0;$i -lt $arr.Length;$i++){
    $arr[$i] = [Convert]::ToByte($clean.Substring($i*2,2),16)
  }
  return $arr
}

# ---- Deterministic DSL ops (MVP minimal set) ----
# Each op appends bytes to the output in order.
function Execute-Plan($plan){
  $ops = $plan.ops
  if(-not $ops){ return ,@() }
  $buf = New-Object System.Collections.Generic.List[byte]
  foreach($op in $ops){
    $name = $op.op
    $args = $op.args

    switch ($name) {
      'gen.repeat' {
        # args: byte(int 0..255), count(uint64)
        if(-not $args.ContainsKey('byte') -or -not $args.ContainsKey('count')){
          throw "gen.repeat requires args.byte and args.count"
        }
        $b = [byte]$args.byte
        $count = [uint64]$args.count
        # chunked append to avoid large allocations
        $chunk = New-Object byte[] ( [Math]::Min(1048576, [int][Math]::Max(1,[int]$count)) )
        for($i=0;$i -lt $chunk.Length;$i++){ $chunk[$i] = $b }
        $remaining = $count
        while($remaining -gt 0){
          $take = [int]([Math]::Min($remaining, [uint64]$chunk.Length))
          $buf.AddRange($chunk[0..($take-1)])
          $remaining -= [uint64]$take
        }
      }
      'write.hex' {
        if(-not $args.ContainsKey('data')){ throw "write.hex requires args.data" }
        $bytes = Hex-To-Bytes $args.data
        $buf.AddRange($bytes)
      }
      default {
        throw "Unsupported op: $name"
      }
    }
  }
  return ,$buf.ToArray()
}

function Do-Restore([string]$seedPath, [string]$outPath){
  if(-not $seedPath){ throw "restore: --seed <path> is required" }
  if(-not $outPath){ throw "restore: --out <path> is required" }
  $s = Read-Json $seedPath
  if(-not $s.core_sha256){ throw "seed missing core_sha256" }
  $coreId = Get-CoreId
  if($coreId -ne $s.core_sha256){ throw "core hash mismatch: seed=$($s.core_sha256) core=$coreId" }

  $bytes = Execute-Plan $s.plan
  if($null -eq $bytes){ $bytes = @() }

  if($s.output_size -ne $bytes.Length){ throw "size mismatch: expected $($s.output_size) got $($bytes.Length)" }

  Write-Bytes $bytes $outPath

  $h = Get-FileHash -LiteralPath $outPath -Algorithm SHA256
  if($h.Hash -ne $s.output_sha256){
    throw "sha256 mismatch: expected $($s.output_sha256) got $($h.Hash)"
  }

  Write-Host "OK restore | out=$outPath | size=$($bytes.Length) | sha256=$($h.Hash)"
}

function Do-Verify([string]$seedPath, [string]$filePath){
  if(-not $seedPath){ throw "verify: --seed <path> required" }
  if(-not $filePath){ throw "verify: --file <path> required" }
  $s = Read-Json $seedPath
  $h = Get-FileHash -LiteralPath $filePath -Algorithm SHA256
  if($h.Hash -ne $s.output_sha256){ throw "verify failed: file sha256=$($h.Hash) != seed sha256=$($s.output_sha256)" }
  if((Get-Item $filePath).Length -ne [int64]$s.output_size){ throw "verify failed: size mismatch" }
  Write-Host "OK verify | file=$filePath"
}

function Try-Compress-MVP([string]$inPath, [string]$seedOut){

  if(-not $seedOut){ throw "compress: --out <seed> required" }

  $coreId = Get-CoreId

  $b0 = $bytes[0]
  $allSame = $true
  for($i=1;$i -lt $bytes.Length;$i++){ if($bytes[$i] -ne $b0){ $allSame = $false; break } }

  $seedObj = $null

    $seedObj = [ordered]@{
      core_sha256   = $coreId
      core_version  = "v0.1.0"
      budget_id     = $budget

      output_sha256 = $shaOut
      regime        = "autonomo"
      plan          = [ordered]@{
        dsl_version = "v0.1.0"
        ops = @(@{ op="gen.repeat"; args=@{ byte=[int]$b0; count=[uint64]$bytes.Length } })
      }

        source_format_hint = "binary"
        notes = "MVP compress: all-same-byte"
      }
    }
  }

    if($bytes.Length -le 4096){

      $seedObj = [ordered]@{
        core_sha256   = $coreId

          source_format_hint = "binary"
          notes = "MVP compress: tiny literal"
        }
      }
    }

# (1) Core Spec

**Nome:** TEIA Deterministic Symbolic Core (MVP)
**Versão:** v0.1.0
**Invariantes:** determinístico, idempotente, sem rede, sem PRNG, sem I/O fora do contrato.
**Contrato do Core:** mesmo binário para `compress` e `restore`; execução pura da seed; validações de tamanho e SHA-256.

**VM/Intérprete (MVP):**

* Sem efeitos colaterais além dos bytes produzidos.
* Serialização/entrada da seed em JSON UTF-8; valida chaves obrigatórias.
* Ordem estrita das operações (lista `plan.ops` executada sequencialmente).
* Execução em **modo streaming** (append em buffer com chunks) para não estourar memória.

**Primitivas (MVP):**

* `gen.repeat(byte,count)` → gera `count` bytes iguais a `byte`.
* `write.hex(data)` → anexa sequência binária definida por string hex canônica.

**Determinismo e Idempotência:**

* `core_sha256` fixado por arquivo do core.
* Planos `plan.ops` executados em ordem; sem fontes de variação.
* `restore` só depende de (`seed`,`core`), e exige `output_size` e `output_sha256`.

**Orçamentos e Scheduler (especificação):**

* O MVP inclui `budget_id` na seed (documentado), preservando espaço para **busca canônica determinística** na fase `compress`.
