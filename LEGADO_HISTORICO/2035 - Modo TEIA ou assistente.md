### Agente Fractal Procedural — instalação (padrão textdoc embed Codex)

```powershell
# TEIA | Agente Fractal Procedural - Instalação/Atualização idempotente
# Compatível com PowerShell 7+. Sem símbolos ocultos; comentários curtos e funcionais.

$Root = 'D:\Teia\TEIA_NUCLEO\offline\nano'
$AgentDir = Join-Path $Root 'agent'
$LogsDir = Join-Path $Root 'logs'
$ConfigPath = Join-Path $AgentDir 'Agent-Fractal-Procedural.config.json'
$AgentScript = Join-Path $Root 'TEIA-Agent-Fractal-Procedural.ps1'

# 1) Pastas
$null = New-Item -ItemType Directory -Path $AgentDir -Force -ErrorAction SilentlyContinue
$null = New-Item -ItemType Directory -Path $LogsDir -Force -ErrorAction SilentlyContinue

# 2) Config JSON (idempotente: recria com defaults seguros; ajuste depois se desejar)
$defaultConfig = @"
{
  "Root": "D:\\Teia\\TEIA_NUCLEO\\offline\\nano",
  "Paths": {
    "BackupsRoot": "D:\\Teia\\TEIA_NUCLEO\\offline\\nano\\backups",
    "LatestPackageZip": "D:\\Teia\\TEIA_NUCLEO\\offline\\nano\\backups\\teia_consolidated_package_20250827_175202.zip",
    "FractalIndex": "D:\\Teia\\TEIA_NUCLEO\\offline\\nano\\fractal_index.json",
    "ManifestDir": "D:\\Teia\\TEIA_NUCLEO\\offline\\nano\\Fractal_Manifests",
    "LogsDir": "D:\\Teia\\TEIA_NUCLEO\\offline\\nano\\logs",
    "AgentLog": "D:\\Teia\\TEIA_NUCLEO\\offline\\nano\\agent\\agent.log",
    "EventsLog": "D:\\Teia\\TEIA_NUCLEO\\offline\\nano\\logs\\dna_events.jsonl"
  },
  "KnownHashes": {
    "LatestPackageZip_SHA256": "1CE56D8B8C02DACF7987F0F5721C146D6F6029D6EF449743292D3ED9EBE7520F",
    "MicroKnowledge_SHA256": "7D919D51B9BAA87A64BD0BFA0F5581E86CE94DA285380A77811BA201A79B2925"
  },
  "Scripts": {
    "Launcher": "D:\\Teia\\TEIA_NUCLEO\\offline\\nano\\TEIA-Fractal-Launch.ps1",
    "RestoreOnDemand": "D:\\Teia\\TEIA_NUCLEO\\offline\\nano\\TEIA-Fractal-Restore-OnDemand.ps1",
    "Preload": "D:\\Teia\\TEIA_NUCLEO\\offline\\nano\\TEIA-Fractal-Preload.ps1",
    "BenchmarkIO": "D:\\Teia\\TEIA_NUCLEO\\offline\\nano\\TEIA-Fractal-Benchmark-IO.ps1",
    "Query": "D:\\Teia\\TEIA_NUCLEO\\offline\\nano\\TEIA-Fractal-Query.ps1"
  },
  "Agent": {
    "MaxRetries": 2,
    "RetryDelaySeconds": 5,
    "StrictSHA": true,
    "FailFast": false
  }
}

Set-Content -Path $ConfigPath -Value $defaultConfig -Encoding UTF8

# 3) Script do Agente
$agentContent = @"
# TEIA-Agent-Fractal-Procedural.ps1
param(
  [ValidateSet('Launch','Restore','Preload','Benchmark','Health','FullCycle')]
  [string] \$Mode = 'Health'
)

# --- Util ---
function Write-Log {
  param([string]\$Msg,[string]\$Level='INFO')
  \$timestamp = (Get-Date).ToString('yyyy-MM-dd HH:mm:ss.fff')
  \$line = "{`"ts`":`"\$timestamp`",`"lvl`":`"\$Level`",`"msg`":`"\$([Regex]::Escape(\$Msg))`"}"
  \$cfg = Get-Config
  Add-Content -Path \$cfg.Paths.AgentLog -Value "[\$timestamp] [\$Level] \$Msg"
  Add-Content -Path \$cfg.Paths.EventsLog -Value \$line
  Write-Host "[\$Level] \$Msg"
}
function Fail { param([string]\$Msg); Write-Log \$Msg 'ERROR'; if((Get-Config).Agent.FailFast){ exit 1 } }

# --- Config / Ambiente ---
\$global:CFG = \$null
function Get-Config {
  if(\$null -eq \$global:CFG){
    \$cfgPath = 'D:\Teia\TEIA_NUCLEO\offline\nano\agent\Agent-Fractal-Procedural.config.json'
    if(-not (Test-Path \$cfgPath)){ throw "Config não encontrada em \$cfgPath" }
    \$global:CFG = Get-Content -Raw -Path \$cfgPath | ConvertFrom-Json
  }
  return \$global:CFG
}
function Assert-Env {
  \$psv = \$PSVersionTable.PSVersion
  if(-not (\$psv.Major -ge 7)){ throw "PowerShell 7+ requerido. Versão atual: \$psv" }
  \$cfg = Get-Config
  foreach(\$p in @(\$cfg.Paths.BackupsRoot,\$cfg.Paths.FractalIndex,\$cfg.Paths.ManifestDir,\$cfg.Paths.LogsDir)){
    if(-not (Test-Path \$p)){ throw "Caminho ausente: \$p" }
  }
  foreach(\$s in (\$cfg.Scripts | Get-Member -MemberType NoteProperty | Select-Object -ExpandProperty Name)){
    \$sp = \$cfg.Scripts.\$s
    if(-not (Test-Path \$sp)){ throw "Script requerido não encontrado: \$s → \$sp" }
  }
  Write-Log "Ambiente OK (PS7+ e caminhos validados)."
}

# --- Util: SHA256 ---
function Get-FileSHA256 { param([string]\$Path)
  if(-not (Test-Path \$Path)){ return \$null }
  (Get-FileHash -Path \$Path -Algorithm SHA256).Hash.ToUpperInvariant()
}
function Verify-PackageSha {
  \$cfg = Get-Config
  \$zip = \$cfg.Paths.LatestPackageZip
  \$expect = \$cfg.KnownHashes.LatestPackageZip_SHA256
  \$hash = Get-FileSHA256 \$zip
  if(-not \$hash){ Fail "Pacote não encontrado: \$zip"; return \$false }
  if(\$cfg.Agent.StrictSHA -and \$hash -ne \$expect){ Fail "SHA256 divergente para pacote: \$hash ≠ \$expect"; return \$false }
  Write-Log "SHA256 OK para pacote: \$hash"
  return \$true
}

# --- Invocações TEIA ---
function Invoke-Launcher {
  \$cfg = Get-Config
  & \$cfg.Scripts.Launcher *>\$null
  if(\$LASTEXITCODE -ne 0){ throw "Launcher falhou (exit \$LASTEXITCODE)" }
  Write-Log "Launcher concluído."
}
function Invoke-Restore { & (Get-Config).Scripts.RestoreOnDemand *>\$null; if(\$LASTEXITCODE -ne 0){ throw "Restore falhou (\$LASTEXITCODE)" }; Write-Log "Restore concluído." }
function Invoke-Preload { & (Get-Config).Scripts.Preload *>\$null; if(\$LASTEXITCODE -ne 0){ throw "Preload falhou (\$LASTEXITCODE)" }; Write-Log "Preload concluído." }
function Invoke-Benchmark { & (Get-Config).Scripts.BenchmarkIO *>\$null; if(\$LASTEXITCODE -ne 0){ throw "Benchmark falhou (\$LASTEXITCODE)" }; Write-Log "Benchmark concluído." }

# --- Health ---
function Test-FractalIndex {
  \$idx = (Get-Config).Paths.FractalIndex
  try {
    \$json = Get-Content -Raw -Path \$idx | ConvertFrom-Json
    if(\$null -eq \$json){ throw "fractal_index.json vazio" }
    Write-Log "fractal_index.json legível (top-level: \$([string]::Join(',',\$json.PSObject.Properties.Name | Select-Object -First 5)))"
    return \$true
  } catch { Fail "Erro ao validar fractal_index.json: \$(\$_.Exception.Message)"; return \$false }
}
function Health {
  Assert-Env
  \$ok1 = Verify-PackageSha
  \$ok2 = Test-FractalIndex
  if(\$ok1 -and \$ok2){ Write-Log "Health OK." } else { Fail "Health com pendências." }
}

# --- Estratégia de Retentativa ---
function Try-Run {
  param([scriptblock]\$Action,[string]\$Name)
  \$cfg = Get-Config
  for(\$i=0; \$i -le \$cfg.Agent.MaxRetries; \$i++){
    try { & \$Action; return \$true }
    catch {
      Write-Log "\$Name falhou tentativa #\$i: \$(\$_.Exception.Message)" 'WARN'
      Start-Sleep -Seconds \$cfg.Agent.RetryDelaySeconds
    }
  }
  Fail "\$Name falhou após retentativas."
  return \$false
}

# --- Orquestração ---
function Run-Launch { Try-Run { Invoke-Launcher } 'Launch' | Out-Null }
function Run-Restore { Try-Run { Invoke-Restore } 'Restore' | Out-Null }
function Run-Preload { Try-Run { Invoke-Preload } 'Preload' | Out-Null }
function Run-Benchmark { Try-Run { Invoke-Benchmark } 'Benchmark' | Out-Null }
function Run-Health { Health }

function Run-FullCycle {
  Run-Health
  Run-Launch
  Run-Preload
  Run-Benchmark
  Write-Log "FullCycle concluído."
}

# --- Main ---
try {
  switch (\$Mode) {
    'Launch'    { Run-Launch }
    'Restore'   { Run-Restore }
    'Preload'   { Run-Preload }
    'Benchmark' { Run-Benchmark }
    'Health'    { Run-Health }
    'FullCycle' { Run-FullCycle }
  }
  exit 0
} catch {
  Fail "Exceção não tratada: \$(\$_.Exception.Message)"
  exit 2
}
"@

Set-Content -Path $AgentScript -Value $agentContent -Encoding UTF8

# 4) Saída amigável
Write-Host "Agente Fractal Procedural instalado/atualizado."
Write-Host "Config: $ConfigPath"
Write-Host "Script: $AgentScript"
```
