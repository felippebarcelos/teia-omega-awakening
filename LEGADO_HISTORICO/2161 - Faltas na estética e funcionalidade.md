1) Watcher local (ponte UI → CLI “teia”)

Cria um serviço PowerShell que observa uma fila (JSONL) e executa teia restore-index/teia restore-scan, validando SHA no final. Isso concretiza o botão “Restaurar (Via Watcher)”.

# TEIA-Restore-Watcher.ps1 — ponte local UI↔CLI (idempotente)
param(
  [string]$Queue = "$env:TEMP\teia_restore_queue.jsonl",
  [string]$Log   = "$env:TEMP\teia_restore_watcher.log"
)
$ErrorActionPreference='Stop'
$seen = New-Object System.Collections.Generic.HashSet[string]
Write-Host "[WATCH] Fila: $Queue"
New-Item -ItemType File -Force -Path $Queue | Out-Null

while ($true) {
  try {
    $lines = Get-Content -LiteralPath $Queue -Raw -ErrorAction Stop -Encoding UTF8 -Force -EA SilentlyContinue -ReadCount 0
    foreach ($ln in ($lines -split "`n")) {
      if([string]::IsNullOrWhiteSpace($ln)){ continue }
      if(-not $seen.Add($ln)){ continue }
      $job = $ln | ConvertFrom-Json -ErrorAction Stop
      $seed = $job.seed; $out=$job.out; $mode=$job.mode; $root=$job.root; $index=$job.index; $maxscan=[int]($job.maxscan ?? 1000000)

      if($mode -eq 'index'){
        teia restore-index -seed $seed -index $index -out $out
      } else {
        teia restore-scan  -seed $seed -root  $root  -out $out -maxscan $maxscan
      }

      $shaSeed = (Get-Content -LiteralPath $seed -Raw | ConvertFrom-Json).sha256.ToLower()
      $shaOut  = (Get-FileHash -LiteralPath $out -Algorithm SHA256).Hash.ToLower()
      if($shaSeed -ne $shaOut){ throw "SHA mismatch: $out" }

      "[OK] $((Get-Date).ToString('s')) mode=$mode out=$out sha=$shaOut" | Add-Content -LiteralPath $Log -Encoding UTF8
    }
  } catch {
    "[ERR] $((Get-Date).ToString('s')) $($_.Exception.Message)" | Add-Content -LiteralPath $Log -Encoding UTF8
  }
  Start-Sleep -Milliseconds 400
}

// store.js — OPFS (File System Access API) com fallback para IndexedDB
export async function saveJson(path, obj){
  const root = await navigator.storage.getDirectory();
  const parts = path.split('/').filter(Boolean);
  let dir = root;
  for(let i=0;i<parts.length-1;i++){ dir = await dir.getDirectoryHandle(parts[i], {create:true}); }
  const file = await dir.getFileHandle(parts.at(-1), {create:true});
  const w = await file.createWritable();
  await w.write(new Blob([JSON.stringify(obj,null,2)], {type:'application/json'}));
  await w.close();
}
export async function loadJson(path, def=null){
  try{
    const root = await navigator.storage.getDirectory();
    const parts = path.split('/').filter(Boolean);
    let dir = root;
    for(let i=0;i<parts.length-1;i++){ dir = await dir.getDirectoryHandle(parts[i]); }
    const file = await dir.getFileHandle(parts.at(-1));
    const data = await (await file.getFile()).text();
    return JSON.parse(data);
  } catch { return def; }
}

worker.js

self.onmessage = async (e) => {
  const { cmd, file } = e.data;
  if(cmd === 'hash'){
    const reader = file.stream().getReader();
    const crypto = self.crypto || globalThis.crypto;
    const chunks = [];
    let read = 0, total = file.size;
    while(true){
      const {value, done} = await reader.read();
      if(done) break;
      read += value.byteLength;
      chunks.push(value);
      self.postMessage({ type:'progress', pct: Math.round((read/total)*100) });
    }
    const blob = new Blob(chunks);
    const buf  = await blob.arrayBuffer();
    const hash = await crypto.subtle.digest('SHA-256', buf);
    const hex  = [...new Uint8Array(hash)].map(b=>b.toString(16).padStart(2,'0')).join('').toLowerCase();
    self.postMessage({ type:'done', hex });
  }
};
