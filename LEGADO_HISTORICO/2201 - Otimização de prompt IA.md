$files = Get-ChildItem -Recurse -File -Path $workingDir | ForEach-Object {
    [PSCustomObject]@{
        FullPath = $_.FullName
        RelativePath = $_.FullName.Substring($workingDir.Length + 1)
        Size = $_.Length
    }
}

# Agrupar em lotes â€490KB
$parts = @()
$currentGroup = @()
$currentSize = 0
$partIndex = 1

foreach ($file in $files) {
    if (($currentSize + $file.Size) -gt $maxPartSizeBytes -and $currentGroup.Count -gt 0) {
        $parts += ,@{
            Index = $partIndex
            Files = $currentGroup
        }
        $partIndex++
        $currentGroup = @()
        $currentSize = 0
    }
    $currentGroup += $file
    $currentSize += $file.Size
}
if ($currentGroup.Count -gt 0) {
    $parts += ,@{
        Index = $partIndex
        Files = $currentGroup
    }
}

# Compactar cada parte em novo ZIP
$logLines = @()
$logLines += "Origem: $sourceZip"
$logLines += "Total de partes geradas: $($parts.Count)"
$logLines += ""

foreach ($part in $parts) {
    $idx = "{0:D3}" -f $part.Index
    $outZip = Join-Path $baseDir "${partPrefix}${idx}.zip"

    # Criar pasta temporĂĄria por parte
    $tempOut = Join-Path $workingDir "out_$idx"
    New-Item -ItemType Directory -Path $tempOut | Out-Null

    # Copiar os arquivos
    foreach ($file in $part.Files) {
        $destPath = Join-Path $tempOut $file.RelativePath
        New-Item -ItemType Directory -Path (Split-Path $destPath) -Force | Out-Null
        Copy-Item -Path $file.FullPath -Destination $destPath
    }

    # Criar o ZIP da parte
    if (Test-Path -LiteralPath $outZip) { Remove-Item $outZip -Force }
    Compress-Archive -Path (Join-Path $tempOut "*") -DestinationPath $outZip

    # Calcular hash e tamanho
    $hash = (Get-FileHash -LiteralPath $outZip -Algorithm SHA256).Hash
    $size = (Get-Item -LiteralPath $outZip).Length

    # Log
    $logLines += "Parte $idx: $outZip"
    $logLines += "  Tamanho: $size bytes"
    $logLines += "  SHA256: $hash"
    $logLines += "  Arquivos incluĂ­dos:"
    $logLines += ($part.Files | ForEach-Object { "    - $_.RelativePath" })
    $logLines += ""
}

# Salvar log final
$logLines | Set-Content -LiteralPath $logPath -Encoding UTF8

Write-Host "`nâïž DivisĂŁo concluĂ­da. Arquivos gerados e log salvo em:"
Write-Host " - $baseDir"
Write-Host " - $logPath`n"
