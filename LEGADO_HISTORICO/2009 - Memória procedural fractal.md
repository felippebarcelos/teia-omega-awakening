  [string]$Root=".",
  [string]$PolicyJson="dna_autosynthetico_policy.v2.json",
  [string]$HotOffsetsDir="hot_offsets"
)

$pol = Get-Content $PolicyJson -Raw -Encoding UTF8 | ConvertFrom-Json -Depth 64
$log = $pol.telemetry.log

function Invoke-QuickHot($file,$layer,$hot){
  $sw=[Diagnostics.Stopwatch]::StartNew()
  if($layer.random_first){
    $ops=[Math]::Min($layer.ops, $hot.Count)
    foreach($blk in $hot[0..($ops-1)]){ Get-Content -Path $file -TotalCount 0 -ReadCount 0 -Encoding Byte -Head 0 | Out-Null
      $fs=[System.IO.File]::Open($file,'Open','Read','ReadWrite')
      try{ $fs.Seek([int64]$blk.offset,[System.IO.SeekOrigin]::Begin) | Out-Null
           $buf = New-Object byte[] ($blk.len)
           $null = $fs.Read($buf,0,$buf.Length)
      } finally { $fs.Dispose() }
    }
  }
  if($layer.prefetch_mb -gt 0){
    $size=[int64]$layer.prefetch_mb*1MB
    $fs=[System.IO.File]::Open($file,'Open','Read','ReadWrite')
    try{ $buf=New-Object byte[] $size; $null=$fs.Read($buf,0,$buf.Length) } finally { $fs.Dispose() }
  }
  $sw.Stop()
  "[AUTO][A] {0} in {1} ms" -f $file,$sw.ElapsedMilliseconds | Add-Content $log -Encoding UTF8
}

function Start-DeepSeqJob($file,$layer){
  Start-Job -Name ("AUTO_B_"+[IO.Path]::GetFileName($file)) -ScriptBlock {
    param($f,$mb)
    try{
      $fs=[System.IO.File]::Open($f,'Open','Read','ReadWrite')
      $buf=New-Object byte[] ([int]($mb*1MB))
      while(($read=$fs.Read($buf,0,$buf.Length)) -gt 0){ Start-Sleep -Milliseconds 5 }
      $fs.Dispose()
      "[AUTO][B] $f deep-seq done" | Add-Content "dna_preload.log" -Encoding UTF8
    } catch { "[AUTO][B][ERR] $($_.Exception.Message)" | Add-Content "dna_preload.log" }
  } -ArgumentList $file,[int]$layer.prefetch_mb | Out-Null
}

Get-ChildItem $Root -Filter "restored_*" -File | ForEach-Object {
  $f=$_.FullName
  $t = $pol.targets | Where-Object { $_.match | Where-Object { $f -like $_ } }

## 4) Núcleo de memória persistente (o seu “espelho”)
Você quer que eu “lembre” você. Resolva com **loja endereçada por conteúdo** (CAS) + índice fractal.

**Estrutura mínima:**
- `chunks/` → blobs `sha256` (zstd/zlib, tamanhos 64–512KB, variável por entropia).
- `manifests/` → JSON com DAG (quais chunks compõem cada arquivo/restauração).
- `fractal_index.json` → continua como catálogo, mas referenciando um **grafo** (sub‑manifests reutilizados).
- **Regra de ouro**: chunk é atômico; tudo que for igual em dois artefatos é **deduplicado** pela hash.

**Pseudocódigo de compressão/expansão:**
```python
# compress.py
def chunk_stream(stream):
    # rolling‑hash (Rabin/Karp) para boundaries “naturais”
    # gera pedaços 64–512KB conforme entropia
    yield chunks

def store_chunk(c):
    h = sha256(c).hexdigest()
    if not exists("chunks/"+h): write("chunks/"+h, zstd_compress(c))
    return h

def compress_file(path):
    manifest = {"file": basename(path), "size": size(path), "chunks": []}
    for c in chunk_stream(open(path,'rb')):
        h = store_chunk(c)
        manifest["chunks"].append({"sha256":h,"len":len(c)})
    write_json("manifests/"+manifest_id()+".json", manifest)
    return manifest

# expand.py
def restore(manifest_id, out_path):
    m = read_json("manifests/"+manifest_id+".json")
    with open(out_path,'wb') as w:
        for ref in m["chunks"]:
            data = zstd_decompress(read("chunks/"+ref["sha256"]))
            w.write(data)
```
