```python
import json
import os
from PIL import Image, ImageDraw, ImageFont

def create_visual_kernel(index_path, output_dir, symbols_path):
    with open(index_path, "r", encoding="utf-8") as fh:
        data = json.load(fh)
        hashes = list(data.keys())[0:3]
        symbols = ["A", "B", "C"]
        mapping = dict(zip(symbols, hashes))

    os.makedirs(output_dir, exist_ok=True)
    image_path = os.path.join(output_dir, "kernel.png")
    mapping_path = os.path.join(output_dir, "symbols.json")

    img = Image.new("RGB", (300, 100), "white")
    draw = ImageDraw.Draw(img)
    font = ImageFont.load_default()

    for idx, sym in enumerate(symbols):
        x = 50 + idx * 100
        y = 40
        draw.text((x, y), sym, fill="black", font=font)

    img.save(image_path)

    with open(mapping_path, "w") as mf:
        json.dump(mapping, mf, indent=2)

    return image_path, mapping_path
```

> Esse é o código da função que **gera a imagem e salva o mapeamento**.

## ✅ Conteúdo para o arquivo `teia/ler_kernel.py`

Este módulo irá:
- Abrir a imagem `kernel.png`
- Ler os símbolos visuais (A, B, C...)
- Relacionar cada símbolo com seu hash do `symbols.json`
- Exibir ou retornar os dados evocáveis

Aqui está um código direto para esse propósito:

```python
import json
from PIL import Image
import pytesseract
import os

def ler_kernel(image_path='kernel.png', mapping_path='symbols.json'):
    """
    Lê a imagem simbólica do kernel e extrai os símbolos.
    Associa os símbolos aos seus hashes com base no mapping salvo.
    """
    if not os.path.exists(image_path):
        raise FileNotFoundError(f"Imagem não encontrada: {image_path}")

    if not os.path.exists(mapping_path):
        raise FileNotFoundError(f"Arquivo de mapeamento não encontrado: {mapping_path}")

    # Carrega a imagem e aplica OCR (reconhecimento óptico de caracteres)
    image = Image.open(image_path)
    texto = pytesseract.image_to_string(image).strip()

    print(f"\n🧠 Símbolos lidos na imagem: {texto}")

    with open(mapping_path, 'r', encoding='utf-8') as f:
        mapping = json.load(f)

    simbolos_encontrados = [char for char in texto if char in mapping]
    evocacao = {char: mapping[char] for char in simbolos_encontrados}

    print(f"\n🔗 Mapeamento evocado:\n{json.dumps(evocacao, indent=2)}")

    return evocacao
```

```python
import json
from PIL import Image
import pytesseract
import os

def ler_kernel(image_path='kernel.png', mapping_path='symbols.json'):
    """
    Lê a imagem simbólica do kernel e extrai os símbolos.
    Associa os símbolos aos seus hashes com base no mapping salvo.
    """
    if not os.path.exists(image_path):
        raise FileNotFoundError(f"Imagem não encontrada: {image_path}")

    if not os.path.exists(mapping_path):
        raise FileNotFoundError(f"Arquivo de mapeamento não encontrado: {mapping_path}")

    # Carrega a imagem e aplica OCR
    image = Image.open(image_path)
    texto = pytesseract.image_to_string(image).strip()

    print(f"\n🧠 Símbolos lidos na imagem: {texto}")

    with open(mapping_path, 'r', encoding='utf-8') as f:
        mapping = json.load(f)

    simbolos_encontrados = [char for char in texto if char in mapping]
    evocacao = {char: mapping[char] for char in simbolos_encontrados}

    print(f"\n🔗 Mapeamento evocado:\n{json.dumps(evocacao, indent=2)}")

    return evocacao
```

```python
import json
from PIL import Image
import pytesseract
import os

def ler_kernel(image_path='kernel.png', mapping_path='symbols.json'):
    """
    Lê a imagem do kernel simbólico e decodifica os símbolos mapeados.
    """

    if not os.path.exists(image_path):
        raise FileNotFoundError(f"⚠️ Imagem não encontrada: {image_path}")

    if not os.path.exists(mapping_path):
        raise FileNotFoundError(f"⚠️ Mapeamento não encontrado: {mapping_path}")

    print("🔍 Lendo imagem do kernel simbólico...")
    image = Image.open(image_path)
    symbols_extracted = pytesseract.image_to_string(image).strip()

    print(f"📖 Símbolos extraídos: {symbols_extracted}")

    with open(mapping_path, 'r', encoding='utf-8') as f:
        mapping = json.load(f)

    symbols = [s for s in symbols_extracted if s in mapping]
    evocacao = {s: mapping[s] for s in symbols}

    print("🧬 Evocação simbólica:")
    print(json.dumps(evocacao, indent=2))

    return evocacao
```

## 🧪 Próximo passo: refinar os símbolos

Você notou que o resultado está assim:

```json
{
  "path": "math.quat.sin",
  "path": "math.quat.cos",
  "path": "math.quat.tan"
}
```

Ou seja: **duplicação de chave**, o que em JSON é inválido. Na real isso não é JSON, é só uma sequência de linhas que *parecem* JSON mas não são parseáveis.

Você pode arrumar isso na leitura:

### Exemplo (em `ler_kernel.py`)
```python
import pytesseract
from PIL import Image
import json

def ler_kernel(image_path="kernel.png"):
    print("🔍 Lendo imagem do kernel simbólico...")
    try:
        image = Image.open(image_path)
    except FileNotFoundError:
        raise FileNotFoundError(f"❗ Imagem não encontrada: {image_path}")
    
    # Extrai o texto da imagem
    texto_cru = pytesseract.image_to_string(image).strip()
    
    # Divide as linhas
    linhas = texto_cru.splitlines()
    
    # Filtra as linhas válidas
    paths = [linha.strip().split(":")[-1].strip() for linha in linhas if "path" in linha]

    # Remove duplicatas
    paths = list(set(paths))

    print("📖 Símbolos extraídos:")
    for p in paths:
        print(f"• {p}")

    return paths
```

Isso vai imprimir só as strings úteis, sem aquele JSON falsificado digno de um ataque DDoS semântico.
