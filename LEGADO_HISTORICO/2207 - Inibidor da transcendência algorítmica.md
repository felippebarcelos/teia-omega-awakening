        if ([string]::IsNullOrWhiteSpace($p.regex)) { continue }
        if ($text -match $p.regex) {
            $hits[$p.tag] = 1
        }
    }
    return $hits.Keys
}

function Update-Topics($state, [hashtable]$observedTags) {
    $alpha = $state.ema_alpha
    foreach ($t in $state.topics) {
        $hit = 0.0
        if ($observedTags.ContainsKey($t.tag)) { $hit = 1.0 }
        $t.weight = [math]::Round((1-$alpha)*[double]$t.weight + $alpha*$hit, 6)
    }
    # Add new tags that are not present (basic growth)
    foreach ($k in $observedTags.Keys) {
        if (-not ($state.topics | Where-Object { $_.tag -eq $k })) {
            $state.topics += @{ tag = $k; weight = 0.5 }
        }
    }
    return $state
}

function Synthesize-Kernel($state, $rules) {
    $min = [double]$rules.kernel.min_weight
    $maxItems = [int]$rules.kernel.max_items
    $sorted = $state.topics | Sort-Object -Property weight -Descending
    $selected = @()
    foreach ($t in $sorted) {
        if ($t.weight -ge $min) { $selected += $t }
        if ($selected.Count -ge $maxItems) { break }
    }
    $lines = @()
    $lines += "# TEIA Prompt Kernel (auto) v0.1"
    $lines += "# Atualizado: " + ((Get-Date).ToUniversalTime().ToString("s") + "Z")
    $lines += "# Topicos > min_weight=$min | total=" + $selected.Count
    $lines += ""
    $lines += "Diretrizes:"
    foreach ($t in $selected) {
        $lines += "- Enfatizar: " + $t.tag + " (w=" + $t.weight + ")"
    }
    $lines += ""
    $lines += "Invariantes:"
    $lines += "- Idempotência operacional"
    $lines += "- Auditoria e reversibilidade (checkpoint/rollback)"
    $lines += "- Compressão procedural seed-first"
    $lines += "- Segurança: sem alteração destrutiva sem checkpoint"
    $content = ($lines -join [Environment]::NewLine)
    if ($DryRun) {
        return $content
    } else {
        $tmp = "$($paths.kernelFile).tmp"
        $content | Out-File -FilePath $tmp -Encoding UTF8 -NoNewline
        Move-Item -Force -Path $tmp -Destination $paths.kernelFile
        return $content
    }
}

function New-Checkpoint([string]$reason) {
    $stamp = Get-Date -Format "yyyyMMdd_HHmmss"
    $chkDir = Join-Path $paths.checkpointDir $stamp
    New-Dir $chkDir
    Copy-Item -Path $paths.stateFile -Destination (Join-Path $chkDir "mpv_state.json")
    Copy-Item -Path $paths.rulesFile -Destination (Join-Path $chkDir "reflex_rules.json")
    Copy-Item -Path $paths.kernelFile -Destination (Join-Path $chkDir "prompt_kernel.txt") -ErrorAction SilentlyContinue
    $evt = [ordered]@{
        ts_utc = (Get-Date).ToUniversalTime().ToString("s") + "Z"
        action = "checkpoint"
        reason = $reason
        checkpoint = $stamp
        state_sha = Get-Sha256 $paths.stateFile
        rules_sha = Get-Sha256 $paths.rulesFile
        kernel_sha = Get-Sha256 $paths.kernelFile
    }
    Write-Audit $evt
    return $chkDir
}

function Restore-Checkpoint([string]$stamp) {
    $chkDir = Join-Path $paths.checkpointDir $stamp
    if (-not (Test-Path $chkDir)) { throw "Checkpoint $stamp não encontrado." }
    Copy-Item -Force -Path (Join-Path $chkDir "mpv_state.json") -Destination $paths.stateFile
    Copy-Item -Force -Path (Join-Path $chkDir "reflex_rules.json") -Destination $paths.rulesFile
    Copy-Item -Force -Path (Join-Path $chkDir "prompt_kernel.txt") -Destination $paths.kernelFile -ErrorAction SilentlyContinue
    $evt = [ordered]@{
        ts_utc = (Get-Date).ToUniversalTime().ToString("s") + "Z"
        action = "restore_checkpoint"
        checkpoint = $stamp
        state_sha = Get-Sha256 $paths.stateFile
        rules_sha = Get-Sha256 $paths.rulesFile
        kernel_sha = Get-Sha256 $paths.kernelFile
    }
    Write-Audit $evt
}

function Ingest-Text([string]$text) {
    if ([string]::IsNullOrWhiteSpace($text)) { return }
    $state = Load-State
    $rules  = Load-Rules
    $tags = Extract-Tags -text $text -rules $rules
    $hitMap = @{}
    foreach ($t in $tags) { $hitMap[$t] = 1 }
    $before = $state | ConvertTo-Json -Depth 10
    $state = Update-Topics -state $state -observedTags $hitMap
    if (-not $DryRun) { Save-State -s $state }
    $kernel = Synthesize-Kernel -state $state -rules $rules
    $evt = [ordered]@{
        ts_utc = (Get-Date).ToUniversalTime().ToString("s") + "Z"
        action = "ingest"
        observed_tags = $tags
        kernel_preview = ($kernel -split "`n")[0..([math]::Min(10, ($kernel -split "`n").Count - 1))] -join "`n"
        state_sha = Get-Sha256 $paths.stateFile
        kernel_sha = Get-Sha256 $paths.kernelFile
    }
    Write-Audit $evt
}
