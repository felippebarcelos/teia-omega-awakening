### 🔧 Código: `puzzle_weave_bot.py` (com múltiplas tentativas)

```python
from selenium import webdriver
from selenium.webdriver.chrome.options import Options
from selenium.webdriver.chrome.service import Service
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
import time, sys

# Caminhos fixos
DRIVER = r"C:\Users\felip\Puzzle\chromedriver-win64\chromedriver.exe"
CHROME = r"C:\Program Files\Google\Chrome\Application\chrome.exe"
URL = ("https://kszeqgxezf5quhzld4nhpasyilhxphclq2peqi5mrn7utxmqhwga."
       "arweave.net/VLJIGuTJewofKx8ad4JYQs93nEuGnkgjrIt_Sd2QPYw")

# Lista de combinações possíveis
combos = [
    "this is the puzzle weaventury12345678",
    "this is the puzzle weaventura12345678",
    "this is the puzzle weaveentry12345678",
    "this is the puzzle weaventory12345678",
    # Adicione novas tentativas aqui ↓
]

# Validação das combinações
for index, phrase in enumerate(combos):
    chars = list(phrase.replace(" ", ""))
    if len(chars) != 32:
        print(f"❌ Combo {index+1} inválido – tem {len(chars)} caracteres: {phrase}")
        continue

    print(f"🚀 Tentando combinação {index+1}: {phrase}")

    # Configurações do Chrome
    opts = Options()
    opts.binary_location = CHROME
    opts.add_argument("--start-maximized")
    service = Service(DRIVER)
    driver = webdriver.Chrome(service=service, options=opts)

    try:
        driver.get(URL)

        inputs = WebDriverWait(driver, 10).until(
            EC.presence_of_all_elements_located((By.CSS_SELECTOR, "input.inputs"))
        )
        if len(inputs) != 32:
            raise RuntimeError(f"⚠️ Esperava 32 inputs, encontrei {len(inputs)}")

        for elem, char in zip(inputs, chars):
            elem.clear()
            elem.send_keys(char)

        unlock = WebDriverWait(driver, 5).until(
            EC.element_to_be_clickable((By.CSS_SELECTOR, "input[type='submit'][value='unlock']"))
        )
        unlock.click()

        # Aguarda resultado
        time.sleep(3)
        print(f"🔍 Tentativa {index+1} executada.")
        input("▶ Pressione Enter para continuar para a próxima...")

    except Exception as e:
        print(f"❌ Erro na tentativa {index+1}: {e}")

    finally:
        driver.quit()
```

1. Cole esse script no seu `puzzle_weave_bot.py`.
2. No bloco `combos = [...]`, adicione todas as variações que quiser tentar (cada uma com exatamente 32 letras/números).
3. Rode no terminal com:
