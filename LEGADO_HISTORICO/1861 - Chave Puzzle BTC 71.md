Script seguro e simbiótico:
```python
from ecdsa import SigningKey, SECP256k1
import hashlib, base58

sk = SigningKey.generate(curve=SECP256k1)
private_key = sk.to_string().hex()

# Gerar chave pública
vk = sk.get_verifying_key()
public_key_bytes = b'\x04' + vk.to_string()

# Hash público -> endereço BTC
sha256_pk = hashlib.sha256(public_key_bytes).digest()
ripemd160 = hashlib.new('ripemd160', sha256_pk).digest()
prefix_pk = b'\x00' + ripemd160
checksum = hashlib.sha256(hashlib.sha256(prefix_pk).digest()).digest()[:4]
btc_address = base58.b58encode(prefix_pk + checksum).decode()

# Codificar chave privada (formato WIF)
wif = base58.b58encode_check(b'\x80' + sk.to_string()).decode()

print("🔐 Chave privada WIF:", wif)
print("🏦 Endereço BTC:", btc_address)
```
