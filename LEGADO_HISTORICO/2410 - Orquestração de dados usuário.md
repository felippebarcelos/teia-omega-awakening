def is_probably_text(path, max_bytes=4096):

def safe_move(src, dest_dir):
    """Move com proteção de colisão de nome e respeita DRY_RUN."""
    os.makedirs(dest_dir, exist_ok=True)
    base = os.path.basename(src)
    target = os.path.join(dest_dir, base)

    if os.path.abspath(src) == os.path.abspath(target):
        return target

    if os.path.exists(target):
        stem, ext = os.path.splitext(base)
        i = 1
        while True:
            candidate = os.path.join(dest_dir, "%s__%d%s" % (stem, i, ext))
            if not os.path.exists(candidate):
                target = candidate
                break
            i += 1

    if DRY_RUN:
        print("[DRY_RUN] moveria:", src, "->", target)
        return target

    shutil.move(src, target)
    return target

## 1. Diagnóstico do erro de `Path` na URL (RAW‑AUTH v3.4)

Pelo que aparece na tua ontologia de caminhos (D:\TEIA raiz estável, WSL de um lado, serviços HTTP/TEIA-Core-Service do outro) e pelas restrições que você mesmo impôs depois (`NUNCA use Join-Path para montar o path da API`, use sempre o caminho absoluto inteiro), o cenário mais consistente é:

### Erro técnico provavel

**Erro central:** colisão de drive / path absoluto sendo “re-encaixado” em cima de outro root.

Algo assim (exemplo ilustrativo):

```text
Certo (esperado pelo RAW-AUTH):
  GET /read?path=D:\TEIA\seed.json&key=...

Errado (o que o agente provavelmente mandou):
  GET /read?path=D:\TEIA\D:\TEIA\seed.json&key=...
                   ^^^^^^^^
```

Ou variações equivalentes, tipo:

- pré-pendando uma raiz fixa (`D:\TEIA\`) **em cima** de um caminho que já vinha absoluto;
- ou montando o path com funções de join que não detectam que o segundo argumento já tem `D:`.

Quando o script RAW‑AUTH v3.4 tenta decodificar isso:

1. Ele faz o `UrlDecode` do `path` e recebe algo como `D:\TEIA\D:\TEIA\seed.json`.
2. Isso cai depois em algo do tipo:

   ```powershell
   Get-Item -LiteralPath $decodedPath
   ```
3. Esse path contém um **segundo `:` no meio do caminho**, que é inválido em Windows:
   - típico erro: *“Illegal characters in path”* ou equivalente,
   - ou falha de resolução de provider dizendo que o drive não existe.

Ou seja: **o bug não foi barra dupla nem “falta de encoding”, foi colisão de drive / root duplicado** causada por montar um path absoluto em cima de outro path absoluto, e depois empacotar isso no parâmetro `path` da URL.

A tua própria regra posterior (“não use Join-Path com caminhos absolutos, passe sempre o caminho absoluto inteiro na URL”) é justamente o remédio direto para essa classe de erro de colisão.

# ===========================
# ENUMERAÇÃO SEGURA (SEM LOOPS)
# ===========================
function Get-SafeFilesInRoot {
    param(
        [string]$Root
    )

    $rootItem = Get-Item -LiteralPath $Root -ErrorAction SilentlyContinue
    if (-not $rootItem -or -not $rootItem.PSIsContainer) {
        return
    }

    $stack = New-Object System.Collections.Stack
    $stack.Push($rootItem)

    while ($stack.Count -gt 0) {
        $dir = $stack.Pop()
        $children = Get-ChildItem -LiteralPath $dir.FullName -Force -ErrorAction SilentlyContinue

        foreach ($child in $children) {
            if ($child.PSIsContainer) {
                # Evita reparse points (symlinks/junctions)
                if (-not ($child.Attributes -band [IO.FileAttributes]::ReparsePoint)) {
                    $stack.Push($child)
                }
            }
            else {
                $child
            }
        }
    }
}

# ===========================
# FASE 1 – MAPA DE ENTROPIA
# ===========================
function Write-EntropyIndex {
    param(
        [string[]]$RootsParam,
        [string]$IndexOutputPath
    )

    $indexDir = Split-Path -Parent $IndexOutputPath
    if (-not (Test-Path -LiteralPath $indexDir)) {
        New-Item -ItemType Directory -Path $indexDir -Force | Out-Null
    }

    $encoding = [System.Text.Encoding]::UTF8
    $stream   = [System.IO.StreamWriter]::new($IndexOutputPath, $false, $encoding)

    try {
        $stream.WriteLine('[')
        $first = $true

        foreach ($root in $RootsParam) {
            $rootItem = Get-Item -LiteralPath $root -ErrorAction SilentlyContinue
            if (-not $rootItem) {
                Write-TeiaLog -Action 'ROOT_NOT_FOUND' -Data @{ root = $root }
                continue
            }

            foreach ($file in Get-SafeFilesInRoot -Root $root) {
                $fullPath = $file.FullName

                # Blindagem: só indexa o que está dentro dos roots explícitos
                if (-not ($fullPath.StartsWith($rootItem.FullName, [System.StringComparison]::OrdinalIgnoreCase))) {
                    continue
                }

                $ext = [System.IO.Path]::GetExtension($fullPath)
                if ($null -ne $ext) {
                    $ext = $ext.ToLowerInvariant()
                }

                $relativePath = $null
                try {
                    $relativePath = $fullPath.Substring($rootItem.FullName.Length).TrimStart('\')
                }
                catch {
                    $relativePath = $file.Name
                }

                $obj = [PSCustomObject]@{
                    root         = $rootItem.FullName
                    relativePath = $relativePath
                    fullPath     = $fullPath
                    name         = $file.Name
                    sizeBytes    = $file.Length
                    extension    = $ext
                    lastWriteUtc = $file.LastWriteTimeUtc.ToString("o")
                }

                $json = $obj | ConvertTo-Json -Depth 5 -Compress

                if ($first) {
                    $first = $false
                    $stream.Write($json)
                }
                else {
                    $stream.Write(',')
                    $stream.Write($json)
                }
            }
        }

        $stream.WriteLine(']')
    }
    finally {
        $stream.Dispose()
    }

    Write-TeiaLog -Action 'INDEX_WRITTEN' -Data @{
        path  = $IndexOutputPath
        roots = $RootsParam
    }
}

    )

        return 'ReservadoTEIA'

    # Artefato Cognitivo – docs, código, configs

# FASE 2 – REORGANIZAÇÃO
# ===========================
function Invoke-TeiaReorg {
    param(
        [string]$IndexInputPath,
        [string]$AtlasRootPath
    )

    if (-not (Test-Path -LiteralPath $IndexInputPath)) {
        Write-TeiaLog -Action 'INDEX_NOT_FOUND' -Data @{ path = $IndexInputPath }
        Write-Host "Arquivo de índice não encontrado: $IndexInputPath"
        return
    }

    if (-not (Test-Path -LiteralPath $AtlasRootPath)) {
        New-Item -ItemType Directory -Path $AtlasRootPath -Force | Out-Null
    }

    $jsonText = Get-Content -LiteralPath $IndexInputPath -Raw -ErrorAction Stop

    try {
        $entries = $jsonText | ConvertFrom-Json -ErrorAction Stop
    }
    catch {
        Write-TeiaLog -Action 'INDEX_PARSE_ERROR' -Data @{
            path  = $IndexInputPath
            error = $_.Exception.Message
        }
        Write-Host "Falha ao interpretar o JSON do índice."
        return
    }

    foreach ($entry in $entries) {
        $fullPath = $entry.fullPath
        if (-not $fullPath) {
            continue
        }

        if (-not (Test-Path -LiteralPath $fullPath)) {
            Write-TeiaLog -Action 'SKIP_NOT_FOUND' -Data @{ path = $fullPath }
            continue
        }

        # Nunca mexe em TEIA_SINTESE / TEIA_ATLAS
        if ($fullPath -like 'D:\TEIA\TEIA_SINTESE\*' -or
            $fullPath -like 'D:\TEIA\TEIA_ATLAS\*') {

            Write-TeiaLog -Action 'SKIP_TEIA_CORE' -Data @{ path = $fullPath }
            continue
        }

        $ext = $entry.extension
        $category = Get-TeiaCategory -FullPath $fullPath -Extension $ext

        if ($category -eq 'ReservadoTEIA') {
            Write-TeiaLog -Action 'SKIP_RESERVADO' -Data @{
                path     = $fullPath
                category = $category
            }
            continue
        }

        $categoryFolder = switch ($category) {
            'ArtefatoCognitivo' { 'Artefatos_Cognitivos' }
            'MateriaBruta'      { 'Materia_Bruta' }
            'LixoEntropico'     { 'Lixo_Entropico' }
            default             { 'Materia_Bruta' }
        }

        $root = $entry.root
        if (-not $root) {
            $root = 'UNKNOWN_ROOT'
        }

        $rootLabel = switch -Wildcard ($root) {
            'C:\Users\Felippe\Documents*' { 'C_Users_Documents' }
            'C:\Users\Felippe\Downloads*' { 'C_Users_Downloads' }
            'D:\*'                        { 'D_Drive' }
            default                       { 'Other' }
        }

        $relativePath = $entry.relativePath
        if (-not $relativePath) {
            $rootFull = $root
            if (-not $rootFull) {
                $rootFull = [System.IO.Path]::GetPathRoot($fullPath)
            }
            $relativePath = $fullPath.Substring($rootFull.Length).TrimStart('\')
        }

        $destBase = Join-Path $AtlasRootPath (Join-Path $categoryFolder $rootLabel)
        $destPath = Join-Path $destBase $relativePath

        $destDir = Split-Path -Parent $destPath
        if (-not (Test-Path -LiteralPath $destDir)) {
            New-Item -ItemType Directory -Path $destDir -Force | Out-Null
            Write-TeiaLog -Action 'WRITE_MKDIR' -Data @{ path = $destDir }
        }

        if (Test-Path -LiteralPath $destPath) {
            Write-TeiaLog -Action 'SKIP_DEST_EXISTS' -Data @{
                source   = $fullPath
                dest     = $destPath
                category = $category
            }
            continue
        }

        Write-TeiaLog -Action 'MOVE' -Data @{
            source   = $fullPath
            dest     = $destPath
            category = $category
        }

        try {
            Move-Item -LiteralPath $fullPath -Destination $destPath -Force
        }
        catch {
            Write-TeiaLog -Action 'ERROR_MOVE' -Data @{
                source   = $fullPath
                dest     = $destPath
                category = $category
                error    = $_.Exception.Message
            }
        }
    }
}
