```powershell
# ╭─────────────────────────────────────────────╮
# │   TEIA Codex Clix Quantum Executor (v1.0)   │
# ╰─────────────────────────────────────────────╯
param(
    [string] $NodeName = "Teia Ajuda Viva Quântica",
    [string] $Backend = "auto",  # auto, ibm_qasm_simulator, ibm_quito, etc
    [int]    $Shots = 2048,
    [string] $IBMQToken = ""
)

Set-StrictMode -Version Latest
$ErrorActionPreference = 'Stop'

Write-Host "[⚙️] Iniciando execução simbiótica autocontida..." -ForegroundColor Cyan

# 1. Preparar ambiente Python/Qiskit
function Ensure-Qiskit {
    $python = Get-Command python -ErrorAction SilentlyContinue
    if (-not $python) {
        throw "Python não encontrado no PATH."
    }

    $qiskitCheck = & python -c "import qiskit; print(qiskit.__version__)" 2>$null
    if (-not $qiskitCheck) {
        Write-Host "[📦] Instalando qiskit..." -ForegroundColor Yellow
        & python -m pip install qiskit qiskit-aer qiskit-ibm-provider --quiet
    }
}

# 2. Verificar e exportar token IBM, se fornecido
function Export-IBMQToken {
    if ($IBMQToken) {
        Write-Host "[🔐] Exportando IBMQ_TOKEN para ambiente..."
        $env:IBMQ_TOKEN = $IBMQToken
    }
}

# 3. Executar ciclo TEIA-Quantum
function Run-QuantumNode {
    $runner = Join-Path $PSScriptRoot "quantum_node\run.ps1"
    if (-not (Test-Path $runner)) {
        throw "Script run.ps1 não encontrado em quantum_node/"
    }
    Write-Host "[🚀] Executando ciclo run.ps1 com backend '$Backend'..."
    & $runner -NodeName $NodeName -Backend $Backend -Shots $Shots
}

# 4. Verificação pós-execução
function Show-LastEvent {
    $logPath = Join-Path $PSScriptRoot "quantum_node\logs\dna_events.jsonl"
    if (Test-Path $logPath) {
        Write-Host "`n🧬 Último Evento Gravado:" -ForegroundColor Green
        Get-Content $logPath | Select-Object -Last 1 | ConvertFrom-Json | Format-List
    } else {
        Write-Host "[⚠️] Nenhum log encontrado ainda." -ForegroundColor Yellow
    }
}

# Execução
Ensure-Qiskit
Export-IBMQToken
Run-QuantumNode
Show-LastEvent
```
